    function min(x, y)
    {
    return (x < y)? x: y;
     
     }
     
   var Minimal_Angle = (h, m) => 
    {
   
    if (h <0 || m < 0 || h >12 || m > 60)
        console.log("Wrong input");
 
    if (h === 12)
    h = 0;
    if (m === 60)
    {
    m = 0;
    h += 1;
    if(h>12)
        h = h-12;
    }
 
    let hour_angle = 0.5 * (h * 60 + m);
    let minute_angle = 6 * m;
    let angle = Math.abs(hour_angle - minute_angle);

ques 3:- Perfect Number Check. Have you heard of Perfect numbers? Now, You are given an integer N, write a program to check whether the given number is a Perfect Number or not.
ans:- 
    var Perfect_Check = (N) => 
    {
        sum = 1;
     
        for (let i=2; i*i<=N; i++)
        {
            if (N%i===0)
            {
                if(i*i!=N)
                    sum = sum + i + N/i;
                else
                    sum=sum+i;
            }
        }
       
        if (sum === N && N !== 1)
        {
            return "YES";
        }
        else{
        return "NO";   
        }        
    }
